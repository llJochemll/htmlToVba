Dim strSQL As String
Dim myDB As DAO.Database
Dim myRST As DAO.Recordset
Dim fso As Object
Dim Fileout As Object
Set myDB = CurrentDb
strSQL = "SELECT * FROM tblPersoneel"
Set myRST = myDB.OpenRecordset(strSQL, dbOpenDynaset, dbSeeChanges)
If myRST.BOF And myRST.EOF Then
Exit Sub
End If
Set fso = CreateObject("Scripting.FileSystemObject")
Set Fileout = fso.CreateTextFile("C:\Users\Jochem\Desktop\new.html", True, True)
Fileout.Write "<!DOCTYPE html><html><head>    <meta name=""author"" content=""Jochem"" />    <meta charset=""UTF-8"">    <title>Crossplatform</title>    <link href=""styleBase.css"" rel=""stylesheet"">    <link href=""styleMonitor.css"" rel=""stylesheet"" id=""varCss""></head><body>"
myRST.MoveFirst
Do While Not myRST.EOF
Fileout.Write "	<a>"
If isNumeric(myRST("naam")) THEN
Fileout.Write str(myRST("naam"))
Else
Fileout.Write myRST("naam")
End If
Fileout.Write "</a>"
myRST.MoveNext
Loop
Fileout.Write "</body></html>"
